  function getParams() {

    const p = new URLSearchParams(window.location.search);

    return {

      practice: p.get('practice') || '',

      score:    parseFloat(p.get('score')) || null,

      label:    (p.get('label') || '').toUpperCase(),

      segment:  (p.get('segment') || '').toLowerCase(),

      city:     p.get('city') || 'Peterborough, ON',

      cityAvg:  parseFloat(p.get('cityavg')) || 59.6,

      c1:       p.get('c1') || '',

      c1score:  parseFloat(p.get('c1score')) || null,

      c2:       p.get('c2') || '',

      c2score:  parseFloat(p.get('c2score')) || null,

    };

  }

  function scoreColor(score) {

    if (score >= 80) return '#00c9a7';

    if (score >= 60) return '#00c9a7';

    if (score >= 50) return '#ffc542';

    return '#ff5f6d';

  }

      return {
